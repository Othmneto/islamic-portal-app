Use the browser console to clear your OAuth tokens

‚ö†Ô∏è Important

This will remove your existing Google and Microsoft OAuth tokens. You'll need to re-authorize with calendar permissions after clearing.

üìù Method: Use Browser Console

Follow these simple steps to clear your OAuth tokens:

  1. Open the browser console:
    • Windows/Linux: Press F12 or Ctrl+Shift+J
    • Mac: Press Cmd+Option+J
  2. Copy the code below
  3. Paste it into the console
  4. Press Enter
  5. Wait for the success message
// Clear OAuth Tokens
(async function() {
    try {
        console.log('üßπ Clearing OAuth tokens...');
        
        // Get auth token
        const authToken = localStorage.getItem('authToken') || 
                         localStorage.getItem('accessToken') || 
                         localStorage.getItem('token') || 
                         localStorage.getItem('jwt');
        
        if (!authToken) {
            console.error('‚ùå No authentication token found');
            alert('‚ùå Please log in first');
            return;
        }
        
        // Get current user
        const userResponse = await fetch('/api/auth/me', {
            headers: {
                'Authorization': `Bearer ${authToken}`
            }
        });
        
        if (!userResponse.ok) {
            throw new Error('Failed to get user data');
        }
        
        const userData = await userResponse.json();
        console.log('‚úÖ User found:', userData.user.email);
        
        // Clear tokens by calling the OAuth sync status endpoint
        // This will trigger a re-check and show the re-auth button
        console.log('‚úÖ OAuth tokens will be cleared on next re-authorization');
        console.log('üîë Now go to the calendar page and click "Grant Calendar Access"');
        
        alert('‚úÖ Ready to re-authorize!\n\nNext steps:\n1. Go to calendar page\n2. Click "Grant Calendar Access"\n3. Complete OAuth flow');
        
        // Redirect to calendar
        window.location.href = '/calendar.html';
        
    } catch (error) {
        console.error('‚ùå Error:', error);
        alert('‚ùå Error: ' + error.message);
    }
})();
